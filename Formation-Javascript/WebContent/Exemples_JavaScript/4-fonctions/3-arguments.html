Fonctions :



Les arguments d'une fonction sont stockés dans un
objet arguments ressemblant à un array.
• arguments[0] est le premier argument de la fonction.
• arguments[i] est le ième argument de la fonction.
• arguments.length donne le nombre d'arguments
passés à la fonction.

En utilisant arguments, il est par exemple possible:
• d'utiliser une fonction avec plus d'arguments qu'elle est déclarée pouvoir recevoir.
• de définir des fonctions avec un nombre variable d'arguments.


   function myConcat(separator) {
   var result = ""; // initialise La String resultat
      // itère à travers Les arguments
      for (var i = 1; i < arguments.length ; i ++) {
      result += arguments[i] + separator; //concatène les arguments et le séparateur'
      }
   return result;
      }
      
      document.write(myConcat(", ","red", "orange", "blue"));
      document.write(myConcat(";  ", "elephant", "giraffe", "lion", "cheetah"));
      document.write(myConcat(". ","sage", "basil " , "oregano", "pepper", " parsley" ));
