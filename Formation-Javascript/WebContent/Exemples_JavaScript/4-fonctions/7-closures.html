CLOSURES :



Une closure est créée lorsque la fonction interne est
rendue disponible, d'une manière ou d'une autre, à un
scope extérieur à la fonction externe. 
    var  pet = function(name) { // La fonction externe définit une variable "name"
    var getName = function() {
    return name;// La fonction interne a accès a la variable "name" de la fonction externe
    }
    return getName; //Retourne la function interne et donc l'expose aux scopes externes
    } ;
    var myPet = pet("Rex");
    myPet (); // Retourne "Rex"
